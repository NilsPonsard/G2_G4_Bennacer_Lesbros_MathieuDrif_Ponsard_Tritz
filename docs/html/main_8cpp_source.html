57 void invadersGeneration(spaceInvaders &SI, const unsigned &height, const unsigned &width)
58 {
59  SI.invadersPos.clear();
60  unsigned Xshift, Yshift(55);
61  for (unsigned i(0); i < (SI.wave - 1) % 4 + 1; ++i)
62  {
63  Xshift = 0;
64  while (Xshift + SI.invaders.entityWidth + (2 * SI.invaders.entityWidth) /*marge min*/ < width)
65  {
66  SI.invadersPos.push_back(make_pair(pos(Xshift, height - 200 - (Yshift * i)), (SI.wave - 1) / 4 + 1));
67  Xshift += 3 * SI.invaders.entityWidth; /*distance entre invaders*/
68  }
69  }
